library(ggplot2)
study <- sample(seq(1,10), replace = T, 100) #1
Y <- 10 + 7*study + round(rnorm(100,0,5)) #
st_score <- data.frame(Y, study)
head(st_score)
st_score %>%
  ggplot(aes(study, Y))+
  geom_point()

st_score %>% 
  lm(Y~study, data=.) -> lm_model
